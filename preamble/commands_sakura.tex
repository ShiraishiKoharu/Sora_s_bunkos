% ───────────────────────────────────────
% SAKURA GRAPHICS
% ───────────────────────────────────────
\newcommand{\sakura}[1][1cm]{%
    \includegraphics[width=#1]{cherry_blossom}%
}

\NewDocumentCommand{\sakurafade}{O{0.5} O{1cm}}{%
    \begin{tikzpicture}
        \node[opacity=#1] {\includegraphics[width=#2]{cherry_blossom}};
    \end{tikzpicture}%
}

\newcommand{\sakuraline}[1][5]{%
    \begin{center}
        \foreach \i in {1,...,#1} {%
                \sakurafade[0.6][0.5cm]\hspace{2mm}%
            }%
    \end{center}
}

\newcommand{\sakurafall}[1][20]{%
    \begin{tikzpicture}[remember picture, overlay]
        \foreach \i in {1,...,#1} {
                \pgfmathsetmacro{\xpos}{5+rnd*90}
                \pgfmathsetmacro{\ypos}{5+rnd*130}
                \pgfmathsetmacro{\rot}{rnd*360}
                \pgfmathsetmacro{\size}{0.2+rnd*0.4}
                \pgfmathsetmacro{\opac}{0.05+rnd*0.15}
                \node[opacity=\opac, rotate=\rot]
                at ($(current page.south west) + (\xpos mm, \ypos mm)$) {
                    \includegraphics[width=\size cm]{cherry_blossom}
                };
            }
    \end{tikzpicture}%
}

% ───────────────────────────────────────
% STORY HEADER (REFACTORED - THE ZEN STYLE)
% ───────────────────────────────────────
\newcommand{\storyheader}[2]{%
    \cleardoublepage
    \phantomsection
    \thispagestyle{empty} % Dọn sạch Header/Footer ở trang mở đầu truyện

    \vspace*{3cm} % Khoảng sụt khổng lồ từ trên xuống tạo sự yên tĩnh

    \begin{center}
        % Tên truyện cực to, gọn gàng
        \parbox{0.85\textwidth}{%
            \centering
            {\titlefont\huge\bfseries\textcolor{sakura_dark}{#1}}%
        }

        \vspace{1cm}

        % CHỈ MỘT điểm nhấn duy nhất, bé bé xinh xinh
        \includegraphics[width=0.8cm]{cherry_blossom.pdf}
    \end{center}

    \vspace{0.5cm}

    % Câu quote lót nằm lệch phải, màu xám thanh lịch
    \begin{flushright}
        \begin{minipage}{0.6\textwidth}
            \flushright
            \small\itshape\textcolor{bunko_gray}{#2}
        \end{minipage}
    \end{flushright}

    \vspace{2cm} % Đẩy khoảng cách an toàn trước khi vào text chính

    \addcontentsline{toc}{chapter}{#1}
    \markboth{#1}{#1}
}

% ───────────────────────────────────────
% STORY FOOTER
% ───────────────────────────────────────
\newcommand{\storyfooter}[1]{%
    \par\vspace{1cm}

    \begin{center}
        \sakurafade[0.3][0.6cm]\quad
        \sakurafade[0.5][1cm]\quad
        \sakurafade[0.3][0.6cm]
    \end{center}

    \ifstrempty{#1}{}{%
        \vspace{0.5cm}
        \begin{flushright}
            \small\textcolor{sakura_dark}{\textit{P.S. #1}}
        \end{flushright}
    }
    \clearpage
}

% ───────────────────────────────────────
% AUTHOR'S NOTE (LIMITED ONLY)
% ───────────────────────────────────────
\newcommand{\authornote}[1]{%
    \vspace{0.8cm}
    \begin{center}
        \begin{tikzpicture}
            \node[
                draw=sakura_dark,
                rounded corners=5pt,
                fill=sakura_light,
                inner sep=8pt,
                text width=0.85\textwidth,
                align=justify
            ] {
                \footnotesize
                \textcolor{sakura_dark}{\textbf{♡ Lời tác giả}}\\[0.3cm]
                \textcolor{bunko_ink}{#1}
            };
        \end{tikzpicture}
    \end{center}
    \vspace{0.8cm}
}
